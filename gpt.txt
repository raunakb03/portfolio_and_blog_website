import React from 'react';

const MyComponent = ({ data }) => {
  return (
    <div>
      {Object.entries(data).map(([key, value]) => {
        // Generate React component based on the value
        let component = null;

        switch (value) {
          case 'text':
            component = <TextComponent key={key} />;
            break;
          case 'image':
            component = <ImageComponent key={key} />;
            break;
          case 'video':
            component = <VideoComponent key={key} />;
            break;
          // Add more cases for other value types

          default:
            component = null;
        }

        return component;
      })}
    </div>
  );
};

export default MyComponent;
